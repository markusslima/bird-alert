Options



clickToHide

Type: Boolean, Default: true

Enables or disables hiding with a click in the box

var birdAlert = new BirdAlert({
	clickToHide: false
});

closeButton

Type: Boolean, Default: true

Hide or show close button

var birdAlert = new BirdAlert({
	closeButton: false
});

autoHide

Type: Boolean, Default: true

Enable or disable hide message automatically

var birdAlert = new BirdAlert({
	autoHide: false
});

duration

Type: Integer, Default: 5000

Time in milliseconds for the duration of the message

var birdAlert = new BirdAlert({
	duration: 1000
});

showAnimation

Type: String, Default: 'flipInX'

Choose from the Animate CSS animations available to show the message

var birdAlert = new BirdAlert({
	showAnimation: 'fadeIn'
});

hideAnimation

Type: String, Default: 'flipOutX'

Choose from the Animate CSS animations available to hide the message

var birdAlert = new BirdAlert({
	hideAnimation: 'fadeOut'
});


inverseInsert

Type: Boolean, Default: false

Enables or disables new messages to be inserted below the old ones

var birdAlert = new BirdAlert({
	inverseInsert: true
});

preventDuplicate

Type: Boolean, Default: false

Blocks duplicate messages

var birdAlert = new BirdAlert({
	preventDuplicate: true
});

position

Type: String, Default: 'top right'

Sets the message position on the screen

var birdAlert = new BirdAlert({
	position: 'bottom center'
});

quantity

Type: Integer, Default: 7

Limit the number of messages displayed

var birdAlert = new BirdAlert({
	quantity: 3
});

agroup

Type: Boolean, Default: true

Enables or disables message grouping by displaying only one message at a time

var birdAlert = new BirdAlert({
	agroup: false
});

width

Type: Integer, Default: 400

Sets the width of the message box

var birdAlert = new BirdAlert({
	width: 800
});

zIndex

Type: Integer, Default: 1000

Sets the width of the message box

var birdAlert = new BirdAlert({
	zIndex: 3000
});

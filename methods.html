Bird Alert

Jquery plugin for displaying beautiful messages on your web system

Methods

.notify()

.notify({title: String, msg: String, className: String});

Method used to display the message

var birdAlert = new BirdAlert();
birdAlert.notify({
	msg: 'Mensage success!!!', 
	title: 'Success', 
	className: 'success'
});

.set()

.set({option: value [, option: value]})

Updates the current object options

var birdAlert = new BirdAlert();
birdAlert.set({
	zIndex: 2100,
	clickToHide: false,
	showAnimation: 'fadeIn',
	autoHide: false
});

.setDefaults()

.set()

Returns all options to their default value

var birdAlert = new BirdAlert({
	clickToHide: false,
	showAnimation: 'fadeIn',
});
birdAlert.setDefaults();
